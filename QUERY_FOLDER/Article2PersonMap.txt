SELECT ?art ?per ?gname ?mname ?fname
WHERE
{
  ?art a bibo:Article .
  ?art vivo:relatedBy ?auth .
  ?auth a vivo:Authorship .
  ?auth vivo:relates ?per .   
  ?per a foaf:Person .
?per obo:ARG_2000028 ?ind .
?ind a vcard:Individual .
 ?ind vcard:hasName ?name .
 ?name vcard:givenName ?gname .
 ?name vcard:familyName ?fname .
 OPTIONAL {?name vivo:middleName ?mname .}
}
